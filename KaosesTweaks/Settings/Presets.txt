        public virtual IEnumerable<ISettingsPreset> GetBuiltInPresets()
        {
            foreach (var preset in base.GetBuiltInPresets())
            {
                yield return preset;
            }

            yield return new MemorySettingsPreset("native_all_off", "native_all_off", "native all off", () => new CustomSettings
            {
                Property1 = false,
                Property2 = true
            });

            yield return new MemorySettingsPreset(Id, "false", "False", () => new CustomSettings
            {
                Property1 = false,
                Property2 = false
            });

            yield return new MemorySettingsPreset(Id, "true", "True", () => new CustomSettings
            {
                Property1 = true,
                Property2 = true
            });
        }
